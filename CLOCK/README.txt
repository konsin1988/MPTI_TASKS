Syntax:
		--clock : print current time
                
		--date : print current date
                
		--clock_dif : print time after (or before, add '-')
                 hours and minutes entered in args
                (two required: hours minutes; format: 12 45)
                
		--date_dif : print date difference (in days)
                betweem current date and date from args
                (three required: day, monthm, year; format - 12 5 2002)
